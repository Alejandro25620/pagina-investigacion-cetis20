        /* ── ACORDEÓN (funciona para todos los botones .btn-expand) ── */
        function toggleCard(btn) {
            const extra = btn.nextElementSibling;
            const isOpen = btn.classList.contains('open');

            document.querySelectorAll('.btn-expand.open').forEach(function(otherBtn) {
                if (otherBtn !== btn) {
                    otherBtn.classList.remove('open');
                    otherBtn.nextElementSibling.classList.remove('visible');
                }
            });

            if (isOpen) {
                btn.classList.remove('open');
                extra.classList.remove('visible');
            } else {
                btn.classList.add('open');
                extra.classList.add('visible');
            }
        }

        (function() {
            const slides = document.querySelectorAll('.hero-slide');
            const dotsContainer = document.getElementById('slideDots');
            let current = 0;
            let timer;
            const quotes = [
                'Aprender hoy abre las puertas de tu futuro.',
                'La prevención es una decisión inteligente que protege tus metas.',
                'Cada buena elección fortalece tu crecimiento académico.',
                'Tu educación vale más que cualquier riesgo momentáneo.'
            ];
            const quoteEl = document.getElementById('slideQuote');
            if (quoteEl) quoteEl.textContent = quotes[0];

            slides.forEach(function(_, i) {
                const dot = document.createElement('button');
                dot.className = 'slide-dot' + (i === 0 ? ' active' : '');
                dot.setAttribute('aria-label', 'Slide ' + (i + 1));
                dotsContainer.appendChild(dot);
            });

            function getDots() { return dotsContainer.querySelectorAll('.slide-dot'); }

            function goTo(next) {
                if (next === current) return;
                clearInterval(timer);

                const prev = current;
                current = next;

                slides[prev].classList.add('leaving');
                slides[prev].addEventListener('animationend', function handler() {
                    slides[prev].classList.remove('active', 'leaving');
                    slides[prev].removeEventListener('animationend', handler);
                });

                slides[current].classList.add('active', 'entering');
                slides[current].addEventListener('animationend', function handler() {
                    slides[current].classList.remove('entering');
                    slides[current].removeEventListener('animationend', handler);
                });

                if (quoteEl) {
                    quoteEl.classList.add('fade');
                    setTimeout(() => {
                        quoteEl.textContent = quotes[current];
                        quoteEl.classList.remove('fade');
                    }, 800);
                }

                getDots().forEach(function(d, i) {
                    d.classList.toggle('active', i === current);
                });
                startTimer();
            }

            function next() {
                goTo((current + 1) % slides.length);
            }

            function startTimer() {
                timer = setInterval(next, 5000);
            }
            startTimer();
        })();

        /* ===== ANIMACIÓN ALTERNADA PARA LAS DOS PRIMERAS CARDS (Introducción y Problema) ===== */
        (function initHorizontalAnimation() {
            const cards = document.querySelectorAll('#introduccion .card.hover-card');
            if (cards.length === 0) return;

            const observer = new IntersectionObserver((entries) => {
                entries.forEach(entry => {
                    if (entry.isIntersecting) {
                        const card = entry.target;
                        if (!card.classList.contains('animate-left') && !card.classList.contains('animate-right')) {
                            const idx = Array.from(cards).indexOf(card);
                            if (idx % 2 === 0) {
                                card.classList.add('animate-left');
                            } else {
                                card.classList.add('animate-right');
                            }
                        }
                        observer.unobserve(card);
                    }
                });
            }, { threshold: 0.2, rootMargin: "0px 0px -20px 0px" });

            cards.forEach(card => {
                observer.observe(card);
            });
        })();
